#!/bin/bash

echo "== RUNNING SCRIPT by Karpenko for TASK #6 =="

if [ $# -lt 1 ]; then
    echo "ERROR: NO PARAMETERS"
    exit 1
fi

output_file=""
count_files=false
params=()

# Парсинг ключей
while [[ "$1" =~ ^- ]]; do
    case "$1" in
        -o)
            shift
            output_file="$1"
            ;;
        -c)
            count_files=true
            ;;
        -h)
            echo "Help: Usage: ./script.sh [-c] [-o output_file] [file ...]"
            exit 0
            ;;
        *)
            echo "Invalid option: $1"
            exit 1
            ;;
    esac
    shift
done

# Оставшиеся параметры считаются файлами
params=("$@")

echo "==== $(date) ===="

if $count_files; then
    echo "Checked files = ${#params[@]}"
fi

output_content=""

for param in "${params[@]}"; do
    status=""
    if [ -d "$param" ]; then
        status="DIRECTORY"
    elif [ ! -e "$param" ]; then
        status="NOT EXIST"
    elif [ -x "$param" ]; then
        status="EXECUTE"
    elif [ -r "$param" ] && [ ! -w "$param" ]; then
        status="READONLY"
    elif [ -w "$param" ]; then
        status="CAN WRITE"
    fi
    echo "$param = $status"
    output_content+="$param = $status\n"
done

if [ -n "$output_file" ]; then
    echo -e "$output_content" > "$output_file"
    if $count_files; then
        echo "Checked files = ${#params[@]}" >> "$output_file"
    fi
fi
